--Question no 6
Give an appropriate example of insert data from subqueries
--ANS:

insert into J2EE(eid, ename, salary, department_id, department_name)
select e.employee_id, e.first_name, e.salary, d.department_id, d.department_name
from employees e
join departments d on d.department_id = e.department_id
where d.department_id = 80;

--Question no 8
Displays the department numbers and average salaries for those departments with a maximum 
salary that is greater than 12000?
--ANS:
select department_id,avg(salary)
from employees
group by department_id 
having max(salary)>12000;
